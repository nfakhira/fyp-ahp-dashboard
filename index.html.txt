        <div class="bg-indigo-900 text-white p-6 rounded-xl shadow-lg text-center">
            <h1 class="text-2xl font-bold">AHP Decision-Making Online Shopping Platform Dashboard </h1>
            <p class="text-indigo-200 mt-2">Selecting Online Shopping Platforms for UTHM Pagoh Students</p>
        </div>

        <div class="grid grid-cols-1 lg:grid-cols-2 gap-6">
            
            <div class="bg-white p-6 rounded-xl shadow border border-gray-200">
                <h2 class="text-xl font-bold text-gray-800 mb-2">1. Define Student Profile</h2>
                <p class="text-sm text-gray-500 mb-4">Because there are 9 criteria, there are 36 comparisons. Leave most as "1 - Equally Important" and only adjust your top priorities to maintain a valid Consistency Ratio (CR < 0.10).</p>
                
                <div id="inputs-container" class="space-y-3 scrollable-inputs pr-2">
                    </div>
            </div>

            <div class="space-y-6">
                
                <div class="bg-white p-6 rounded-xl shadow border border-gray-200">
                    <h2 class="text-xl font-bold text-gray-800 mb-4">2. Calculated Priority Weights</h2>
                    <div id="weights-container" class="grid grid-cols-2 gap-x-4 gap-y-2 mb-4"></div>
                    
                    <div class="p-3 rounded-lg flex justify-between items-center mt-4" id="cr-box">
                        <span class="font-semibold text-gray-700">Consistency Ratio (CR):</span>
                        <span class="font-bold text-lg" id="cr-value">0.000</span>
                    </div>
                </div>

                <div class="bg-white p-6 rounded-xl shadow border border-gray-200">
                    <h2 class="text-xl font-bold text-gray-800 mb-4">3. Final Platform Ranking</h2>
                    <p class="text-sm text-gray-500 mb-4">Dynamically calculating the best platform.</p>
                    <div id="ranking-container" class="space-y-4"></div>
                </div>

    <script>
        // --- 1. DATA SETUP (9 CRITERIA) ---
        const criteria = [
            "Price & Promo", "Shipping", "Trust & Security", 
            "User Interface", "Return & Refund", "Payment Options", 
            "Cust. Service", "Product Quality", "Social Influence"
        ];
        
        // Generate the 36 pairs automatically
        const pairs = [];
        for(let i=0; i<9; i++) {
            for(let j=i+1; j<9; j++) {
                pairs.push([i, j]);
            }
        }
        
        const platforms = ['Shopee', 'Lazada', 'TikTok Shop', 'Temu', 'Shein', 'Facebook Marketplace', 'eBay', 'Amazon'];
        
        // Competitive scores out of 5.0 so the winner changes based on priorities
        const rawScores = {
            'Shopee': [4.8, 4.5, 3.5, 4.0, 3.5, 4.8, 3.8, 3.8, 4.2],
            'Lazada': [3.8, 4.0, 4.7, 4.2, 4.5, 4.0, 4.0, 4.5, 3.5],
            'TikTok Shop': [4.0, 3.8, 3.2, 4.8, 3.2, 3.5, 4.5, 3.5, 4.9],
            'Temu': [4.9, 3.5, 2.5, 3.5, 3.0, 3.5, 2.8, 2.5, 3.5],
            'Shein': [4.2, 3.2, 3.0, 4.5, 3.0, 3.2, 3.0, 3.2, 4.5],
            'Facebook Marketplace': [4.0, 2.5, 2.0, 3.0, 2.0, 2.5, 2.5, 3.0, 4.0],
            'eBay': [3.5, 3.5, 4.2, 3.0, 3.8, 4.5, 3.5, 4.0, 2.5],
            'Amazon': [3.0, 4.8, 4.8, 4.0, 4.8, 4.5, 4.2, 4.7, 3.0]
        };

        // Calculate Column Sums automatically
        const colSums = Array(9).fill(0);
        platforms.forEach(p => {
            for(let i=0; i<9; i++) colSums[i] += rawScores[p][i];
        });
        
        const normalizedScores = {};
        platforms.forEach(p => {
            normalizedScores[p] = rawScores[p].map((score, i) => score / colSums[i]);
        });

        const scaleOptions = [
            {val: 9, label: "9 - Extremely More"}, {val: 7, label: "7 - Very Strongly"},
            {val: 5, label: "5 - Strongly More"}, {val: 3, label: "3 - Moderately More"},
            {val: 1, label: "1 - Equally Important"},
            {val: 1/3, label: "1/3 - Moderately Less"}, {val: 1/5, label: "1/5 - Strongly Less"},
            {val: 1/7, label: "1/7 - Very Strongly Less"}, {val: 1/9, label: "1/9 - Extremely Less"}
        ];

        // --- 2. UI GENERATION ---
        const inputContainer = document.getElementById('inputs-container');
        pairs.forEach((pair, idx) => {
            let div = document.createElement('div');
            div.className = "flex justify-between items-center text-xs py-1 border-b border-gray-100 last:border-0";
            
            let select = `<select id="pair-${idx}" class="border border-gray-300 rounded p-1 w-40 text-center bg-gray-50 focus:outline-none focus:ring-1 focus:ring-indigo-500" onchange="calculateAHP()">`;
            
            scaleOptions.forEach(opt => {
                let selected = (opt.val === 1) ? "selected" : "";
                select += `<option value="${opt.val}" ${selected}>${opt.label}</option>`;
            });
            select += `</select>`;
            
            div.innerHTML = `
                <span class="font-medium w-1/3 text-right pr-2 truncate text-gray-700" title="${criteria[pair[0]]}">${criteria[pair[0]]}</span> 
                ${select} 
                <span class="font-medium w-1/3 text-left pl-2 truncate text-gray-700" title="${criteria[pair[1]]}">${criteria[pair[1]]}</span>
            `;
            inputContainer.appendChild(div);
        });

        // --- 3. AHP MATH ENGINE ---
        function calculateAHP() {
            let n = 9;
            let matrix = Array(n).fill(0).map(() => Array(n).fill(1));
            
            // Build Matrix
            pairs.forEach((pair, idx) => {
                let val = parseFloat(document.getElementById(`pair-${idx}`).value);
                matrix[pair[0]][pair[1]] = val;
                matrix[pair[1]][pair[0]] = 1 / val;
            });

            // Normalize & Get Weights
            let cSums = Array(n).fill(0);
            for(let c=0; c<n; c++) {
                for(let r=0; r<n; r++) cSums[c] += matrix[r][c];
            }

            let weights = Array(n).fill(0);
            for(let r=0; r<n; r++) {
                let rowSum = 0;
                for(let c=0; c<n; c++) {
                    rowSum += (matrix[r][c] / cSums[c]);
                }
                weights[r] = rowSum / n;
            }

            // Consistency Ratio (RI for 9 criteria is 1.45)
            let lambdaMax = 0;
            for(let i=0; i<n; i++) lambdaMax += (cSums[i] * weights[i]);
            let ci = (lambdaMax - n) / (n - 1);
            let cr = ci / 1.45;

            updateUI(weights, cr);
        }

        function updateUI(weights, cr) {
            // Update CR Box
            let crBox = document.getElementById('cr-box');
            let crVal = document.getElementById('cr-value');
            crVal.innerText = cr.toFixed(3);
            if(cr <= 0.10) {
                crBox.className = "p-3 rounded-lg flex justify-between items-center bg-green-100 text-green-800 border border-green-300";
            } else {
                crBox.className = "p-3 rounded-lg flex justify-between items-center bg-red-100 text-red-800 border border-red-300";
            }

            // Update Weights Text
            let wContainer = document.getElementById('weights-container');
            wContainer.innerHTML = '';
            criteria.forEach((c, i) => {
                wContainer.innerHTML += `<div class="flex justify-between text-xs p-1 bg-gray-50 rounded border border-gray-100"><span class="text-gray-600 truncate mr-2">${c}</span> <span class="font-bold text-indigo-700">${(weights[i]*100).toFixed(1)}%</span></div>`;
            });

            // Calculate Global Scores
            let finalScores = [];
            platforms.forEach(p => {
                let score = 0;
                for(let i=0; i<9; i++) {
                    score += (normalizedScores[p][i] * weights[i]);
                }
                finalScores.push({name: p, score: score});
            });

            // Sort and Render Bars
            finalScores.sort((a, b) => b.score - a.score);
            let topScore = finalScores[0].score;
            
            let rContainer = document.getElementById('ranking-container');
            rContainer.innerHTML = '';
            finalScores.forEach((item, index) => {
                let barWidth = (item.score / topScore) * 100;
                let color = index === 0 ? "bg-indigo-600" : (index === 1 ? "bg-indigo-400" : "bg-gray-300");
                let rankLabel = index === 0 ? "🏆 #1 " : "";
                
                rContainer.innerHTML += `
                    <div>
                        <div class="flex justify-between text-sm font-medium mb-1">
                            <span class="${index===0?'text-indigo-800 font-bold':''}">${rankLabel}${item.name}</span>
                            <span>${item.score.toFixed(3)}</span>
                        </div>
                        <div class="w-full bg-gray-100 rounded-full h-2">
                            <div class="${color} h-2 rounded-full bar-fill" style="width: ${barWidth}%"></div>
                        </div>
                    </div>
                `;
            });
        }

        // Initialize on load
        calculateAHP();
    </script>
